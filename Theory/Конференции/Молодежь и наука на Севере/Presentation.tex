\begin{frame}
\frametitle{ Общая формулировка задачи }
 
\begin{equation}\label{ivanovsky-eq1}
	u' = \beta \ddot{u} - \gamma u(t - \tau') + F(u),
\end{equation}	

\begin{gather}\label{ivanovsky-eq2}	
	u'(0, t) \, = 0, \\
	u'(1, t) \, = \alpha\,u(x_0, t-\tau). \nonumber
\end{gather}

$$ \alpha \in \mathbb{R}, \quad \beta, \gamma > 0, \quad \tau \geqslant 0, \quad x_0 \in [0, 1]. $$

\begin{frame}
\frametitle{Нормальная форма}

\begin{equation}
	u(x,t) = \sum_{j=1}^{\infty} \varepsilon^\frac{j+1}{2}u_j (t,s,x), s = \varepsilon t
\end{equation}

\begin{equation}
	u_{0}(x,t) = z(s)e^{i \omega t} w(x) + \overline{z(s)}e^{-i \omega t}\overline{w(x)}
\end{equation}

\begin{equation}
	\begin{aligned}
		\sqrt{\varepsilon}:& \dot{u}_{0} = u_{0}'' + u_{0} \\
		\varepsilon:& \dot{u}_{1} = u_{1}'' + u_{1} + f( u_{0}) \\
		\varepsilon^{3/2}:& \dot{u}_{2} = u_{2}'' + u_{2} + g(u_{0},u_{1}) \\
	\end{aligned}
\end{equation}

\end{frame}
\begin{frame}\frametitle{Нормальная форма}
	\begin{equation} 
		\label{norm_form}
		\dot{z} = \phi z + d z |z|^2
	\end{equation}
\begin{rustheorem}
	При $ Re(\phi)>0,\, Re(d)<0 \,\exists \varepsilon_0 > 0 \,\, \forall \varepsilon \in (0,\varepsilon_0] $
	наблюдается экспоненциально-орбитально устойчивый цикл, 
	асимптотика которого описывается формулой \ref{norm_form}, в которой
	$z(s) = \sqrt{-\frac{Re(\phi)}{Re(d)}} \exp{\left(i \left(Im(\phi) -
	 \frac{Im(d)Re(\phi)}{Re(d)} \right) s + i\gamma \right)} $
\end{rustheorem}
\end{frame}
\begin{frame}
\frametitle{Квазианалитическое решение}
Положим 
\begin{equation}\label{kuksenok-eq2}
g = - \gamma + i \omega , 
\end{equation}
Тогда
\begin{equation}\label{kuksenok-eq3}
(\ref{ivanovsky-eq8}, \ref{ivanovsky-eq9}) \quad \rightarrow \quad
\left\{
\begin{matrix}
    g^{1/2}\sh{g^{1/2}}-\alpha \exp{(-i \omega \tau)} \ch{g^{1/2}} = 0 & \mbox{if } \tau \neq 0 \\
    g^{1/2}\sh{g^{1/2}}-\alpha  \ch{g^{1/2}} = 0 & \mbox{if } \tau = 0
\end{matrix}
\right.
\end{equation}
$$ \omega \in \mathbb{R}$$
\end{frame}
\begin{frame}
\frametitle{Квазианалитическое решение}

\begin{equation}\label{kuksenok-eq4}
\omega = -i(\gamma + g ), 
\end{equation}
Где $g$ может быть найдено численно из   (\ref{kuksenok-eq3})
\end{frame}
